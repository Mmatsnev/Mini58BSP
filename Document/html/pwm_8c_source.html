36 uint32_t PWM_ConfigOutputChannel (PWM_T *pwm,
37  uint32_t u32ChannelNum,
38  uint32_t u32Frequency,
39  uint32_t u32DutyCycle)
40 {
41  uint32_t i = SystemCoreClock / u32Frequency;
42  uint8_t u8Divider = 1, u8Prescale = 0xFF;
43  uint16_t u16CNR = 0xFFFF;
44 
45  for(; u8Divider < 17; u8Divider <<= 1) // clk divider could only be 1, 2, 4, 8, 16
46  {
47  i = (SystemCoreClock / u32Frequency) / u8Divider;
48  // If target value is larger than CNR * prescale, need to use a larger divider
49  if(i > (0x10000 * 0x100))
50  continue;
51 
52  // CNR = 0xFFFF + 1, get a prescaler that CNR value is below 0xFFFF
53  u8Prescale = (i + 0xFFFF)/ 0x10000;
54 
55  // u8Prescale must at least be 2, otherwise the output stop
56  if(u8Prescale < 3)
57  u8Prescale = 2;
58 
59  i /= u8Prescale;
60 
61  if(i <= 0x10000)
62  {
63  if(i == 1)
64  u16CNR = 1; // Too fast, and PWM cannot generate expected frequency...
65  else
66  u16CNR = i;
67  break;
68  }
69 
70  }
71  // Store return value here 'cos we're gonna change u8Divider & u8Prescale & u16CNR to the real value to fill into register
72  i = SystemCoreClock / (u8Prescale * u8Divider * u16CNR);
73 
74  u8Prescale -= 1;
75  u16CNR -= 1;
76  // convert to real register value
77  if(u8Divider == 1)
78  u8Divider = 4;
79  else if (u8Divider == 2)
80  u8Divider = 0;
81  else if (u8Divider == 4)
82  u8Divider = 1;
83  else if (u8Divider == 8)
84  u8Divider = 2;
85  else // 16
86  u8Divider = 3;
87 
88  // every two channels share a prescaler
89  PWM->CLKPSC = (PWM->CLKPSC & ~(PWM_CLKPSC_CLKPSC01_Msk << ((u32ChannelNum >> 1) * 8))) | (u8Prescale << ((u32ChannelNum >> 1) * 8));
90  PWM->CLKDIV = (PWM->CLKDIV & ~(PWM_CLKDIV_CLKDIV0_Msk << (4 * u32ChannelNum))) | (u8Divider << (4 * u32ChannelNum));
91  PWM->CTL = (PWM->CTL & ~PWM_CTL_CNTTYPE_Msk) | (PWM_CTL_CNTMODE0_Msk << ((4 * u32ChannelNum)));
92 
93  if(u32DutyCycle == 0)
94  *((__IO uint32_t *)((((uint32_t) & ((pwm)->CMPDAT0)) + u32ChannelNum * 4))) = 0;
95  else
96  *((__IO uint32_t *)((((uint32_t) & ((pwm)->CMPDAT0)) + u32ChannelNum * 4))) = u32DutyCycle * (u16CNR + 1) / 100 - 1;
97 
98  *((__IO uint32_t *)((((uint32_t) & ((pwm)->PERIOD0)) + (u32ChannelNum) * 4))) = u16CNR;
99 
100  return(i);
101 }
